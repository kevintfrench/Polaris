SELECT P.Barcode, PR.PatronFirstLastName, B.BrowseTitle, IC.OriginalCheckOutDate, IC.Renewals AS "Renewal Taken" , CIR.RenewalLimit, o.Name as 'Item Assigned Branch' 
FROM polaris.polaris.CircItemRecords CIR WITH (NOLOCK) JOIN polaris.polaris.ItemCheckouts IC WITH (NOLOCK) ON CIR.ItemRecordID = IC.ItemRecordID JOIN polaris.polaris.BiblioGraphicRecords B WITH (NOLOCK) ON B.BibliographicRecordID = CIR.AssociatedBibRecordID JOIN polaris.polaris.PatronRegistration PR WITH (NOLOCK) ON PR.PatronID = IC.PatronID JOIN polaris.polaris.Patrons P WITH (NOLOCK) ON P.PatronID = PR.PatronID join Polaris.Polaris.Organizations o on CIR.AssignedBranchID = o.OrganizationID WHERE IC.Renewals >  CIR.RenewalLimit and o.OrganizationID = @org and CIR.ItemStatusID = 2 ORDER BY IC.Renewals desc, CIR.RenewalLimit, IC.OriginalCheckOutDate

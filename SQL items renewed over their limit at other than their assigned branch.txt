SELECT P.Barcode, PR.PatronFirstLastName, O1.Abbreviation AS "PatBr", O2.Abbreviation AS "ItemBr", B.BrowseTitle, IC.OriginalCheckOutDate, IC.Renewals AS "Renewal Taken" ,CIR.RenewalLimit FROM CircItemRecords CIR WITH (NOLOCK) JOIN ItemCheckouts IC WITH (NOLOCK) ON CIR.ItemRecordID = IC.ItemRecordID JOIN BiblioGraphicRecords B WITH (NOLOCK) ON B.BibliographicRecordID = CIR.AssociatedBibRecordID JOIN PatronRegistration PR WITH (NOLOCK) ON PR.PatronID = IC.PatronID JOIN Patrons P WITH (NOLOCK) ON P.PatronID = PR.PatronID JOIN Organizations O1  WITH (NOLOCK) ON O1.OrganizationID = IC.OrganizationID JOIN Organizations O2  WITH (NOLOCK) ON O2.OrganizationID = CIR.AssignedBranchID WHERE IC.Renewals >  CIR.RenewalLimit AND O1.Abbreviation <> O2.Abbreviation  --comment out this line to see all items AND O1.OrganizationID = 49  --change this to the OrgID of choice or comment out the entire line to see all.
ORDER BY PR.PatronFirstLastName, CIR.RenewalLimit, IC.Renewals, IC.OriginalCheckOutDate
